async function handleFiles(files) { if (files.length === 0) return; // 初始化 const zip = new JSZip(); zipBlob = null; logList.innerHTML = ""; logArea.style.display = "none"; downloadBtn.style.display = "none"; processArea.style.display = "block"; const total = files.length; for (let i = 0; i < total; i++) { const file = files[i]; const fileNameWithExt = file.name; const fileNameOnly = fileNameWithExt.split('.').slice(0, -1).join('.').trim(); // 更新進度 statusText.innerText = `正在處理: ${fileNameWithExt}`; counterText.innerText = `${i + 1} / ${total}`; progressBar.style.width = `${((i + 1) / total) * 100}%`; // 1. 檢查檔名：數字 + 底線 const namePattern = /^[0-9_]+$/; if (!namePattern.test(fileNameOnly)) { addLog(fileNameWithExt, `檔名格式錯誤 (含有數字與底線以外的字元: "${fileNameOnly}")`); continue; } try { // 2. AI 去背 // 這裡會攔截環境與 AI 處理錯誤 let noBgBlob; try { noBgBlob = await imglyRemoveBackground(file); } catch (aiErr) { // 區分是環境問題還是檔案問題 if (window.crossOriginIsolated === false) { throw new Error("安全環境未啟動 (請確認 coi-serviceworker.js 是否正確安裝)"); } else { throw new Error("AI 無法去背此圖 (圖檔可能過大或格式特殊)"); } } // 3. 縮放 800x800 const finalBlob = await resizeTo800(noBgBlob); // 4. 加入打包 zip.file(`${fileNameOnly}.png`, finalBlob); } catch (err) { // 捕捉所有流程中的異常並顯示具體原因 addLog(fileNameWithExt, err.message); } } statusText.innerText = "✅ 處理結束"; if (Object.keys(zip.files).length > 0) { zipBlob = await zip.generateAsync({ type: "blob" }); downloadBtn.style.display = "block"; } else { statusText.innerText = "❌ 無任何檔案符合規範"; } } // 統一的錯誤顯示函式 function addLog(filename, reason) { logArea.style.display = "block"; const div = document.createElement('div'); div.className = "log-item"; div.innerHTML = `❌ 跳過： ${filename}
原因：${reason}`; logList.appendChild(div); }
